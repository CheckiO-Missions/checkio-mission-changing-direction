def changing_direction(elements: list[int]) -> int:

    dir = count = 0
    for a, b in zip(elements, elements[1:]):
        dir2 = a - b
        if dir2:
            if dir2 * dir < 0:
                count += 1
            dir = dir2

    return count

    return sum(x * y < 0 for x, y in zip(d, d[1:]))
